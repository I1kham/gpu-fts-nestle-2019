Per mettere in autorun degli script usando systemd:
	https://learn.sparkfun.com/tutorials/how-to-run-a-raspberry-pi-program-on-startup#method-3-systemd


sudo nano /lib/systemd/system/rheaHotspotSetup.service
	[Unit]
	Description=Modifica le impostazioni dell'hotspot wifi cambiandone il nome e la pwd eseguendo uno script py
	After=multi-user.target

	[Service]
	WorkingDirectory=/home/pi/
	User=pi
	ExecStart=/usr/bin/python2 /home/pi/Desktop/gpu-fts-nestle-2019/bin/rasPI_rheaBoot.py

	[Install]
	WantedBy=multi-user.target


sudo nano /lib/systemd/system/rheaStartRasPIESAPI.service
	[Unit]
	Description=Start rhea bootloader for rasPI prg 
	After=network.target

	[Service]
	WorkingDirectory=/home/pi/rhea/gpu-fts-nestle-2019/bin
	User=pi
	ExecStart=/home/pi/rhea/gpu-fts-nestle-2019/bin/RASPI_RELEASE_rasPIBootLoader

	[Install]
	WantedBy=network.target




sudo systemctl daemon-reload
